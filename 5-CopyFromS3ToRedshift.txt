SELECT COUNT(*)
FROM public.nation n 
INNER JOIN public.customer c on n.n_nationkey = c.c_nationkey
INNER JOIN public.orders o on c.c_custkey = o.o_custkey
INNER JOIN public.lineitem li on o.o_orderkey = li.l_orderkey
WHERE n.n_name = 'CANADA'
